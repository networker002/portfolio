import fastapi
import sqlite3

app = fastapi.FastAPI()

@app.get("/")
def show() -> list:
    conn = sqlite3.connect('my_database.db')
    cursor = conn.cursor()
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS products (
            id INTEGER PRIMARY KEY,
            name TEXT,
            price REAL
        )
    ''')

    data = ('Laptop', 1500.0)
    cursor.execute(
        "INSERT INTO products (
            name,price) VALUES (?, ?)", data)

    conn.commit()
    cursor.execute("SELECT * FROM products")
    rows = cursor.fetchall()
    conn.close()

    return rows
